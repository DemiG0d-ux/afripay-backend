const sdk = require('node-appwrite');

// This is the main entry point for our consolidated function
module.exports = async (req, res) => {
  // --- Setup ---
  const client = new sdk.Client()
    .setEndpoint(req.variables.APPWRITE_FUNCTION_ENDPOINT)
    .setProject(req.variables.APPWRITE_FUNCTION_PROJECT_ID)
    .setKey(req.variables.APPWRITE_API_KEY);

  const databases = new sdk.Databases(client);

  try {
    // --- Step 1: Get the data from the Flutter app ---
    // The 'type' will tell us what kind of transaction this is.
    const { type, amount, currency, details } = JSON.parse(req.payload);
    const senderId = req.variables.APPWRITE_FUNCTION_USER_ID;

    if (!type || !amount || amount <= 0) {
      throw new Error("Invalid transaction type or amount.");
    }

    // --- Step 2: Get the sender's document to check their balance ---
    const senderDoc = await databases.getDocument('afripay_db', 'users', senderId);
    const balanceField = currency === 'GHS' ? 'balanceGHS' : 'balanceNGN';

    // --- Step 3: Check for sufficient funds ---
    if (senderDoc[balanceField] < amount) {
      throw new Error("Insufficient funds.");
    }

    const newSenderBalance = senderDoc[balanceField] - amount;

    // --- Step 4: Use a switch statement to handle different transaction types ---
    switch (type) {
      case 'p2p-transfer':
        // Logic for Peer-to-Peer Transfer
        const { recipientId } = details;
        if (!recipientId) throw new Error("Recipient ID is required for transfers.");
        
        const recipientDoc = await databases.getDocument('afripay_db', 'users', recipientId);
        const newRecipientBalance = recipientDoc[balanceField] + amount;

        await Promise.all([
          databases.updateDocument('afripay_db', 'users', senderId, { [balanceField]: newSenderBalance }),
          databases.updateDocument('afripay_db', 'users', recipientId, { [balanceField]: newRecipientBalance }),
          databases.createDocument('afripay_db', 'transactions', sdk.ID.unique(), { description: `Transfer to ${recipientDoc.name}`, amount, type: 'debit', status: 'Completed', userId: senderId }, [sdk.Permission.read(sdk.Role.user(senderId))]),
          databases.createDocument('afripay_db', 'transactions', sdk.ID.unique(), { description: `Transfer from ${senderDoc.name}`, amount, type: 'credit', status: 'Completed', userId: recipientId }, [sdk.Permission.read(sdk.Role.user(recipientId))]),
        ]);
        break;

      case 'fund-susu':
        // Logic for Funding a Susu Plan
        const { planId } = details;
        if (!planId) throw new Error("Plan ID is required for funding.");

        const planDoc = await databases.getDocument('afripay_db', 'savings_plans', planId);
        const newPlanBalance = planDoc.currentBalance + amount;

        await Promise.all([
          databases.updateDocument('afripay_db', 'users', senderId, { [balanceField]: newSenderBalance }),
          databases.updateDocument('afripay_db', 'savings_plans', planId, { 'currentBalance': newPlanBalance }),
          databases.createDocument('afripay_db', 'transactions', sdk.ID.unique(), { description: `Funding for Ajo/Susu: ${planDoc.planName}`, amount, type: 'debit', status: 'Completed', userId: senderId }, [sdk.Permission.read(sdk.Role.user(senderId))]),
        ]);
        break;

      case 'pay-bill':
        // Logic for Paying a Bill
        const { biller, customerId } = details;
        if (!biller || !customerId) throw new Error("Biller details are required.");
        
        // (SIMULATION) In a real app, you'd call Paystack here.
        console.log(`Simulating payment of ${amount} for ${biller} to customer ${customerId}.`);

        await Promise.all([
          databases.updateDocument('afripay_db', 'users', senderId, { [balanceField]: newSenderBalance }),
          databases.createDocument('afripay_db', 'transactions', sdk.ID.unique(), { description: `${biller} Payment`, amount, type: 'debit', status: 'Completed', userId: senderId }, [sdk.Permission.read(sdk.Role.user(senderId))]),
        ]);
        break;

      default:
        throw new Error("Unknown transaction type.");
    }

    res.json({ success: true, message: 'Transaction successful!' });

  } catch (error) {
    console.error("Transaction failed:", error);
    res.json({ success: false, message: error.message }, 400);
  }
};
