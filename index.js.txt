const sdk = require('node-appwrite');

// This is the main entry point for the transfer function
module.exports = async (req, res) => {
  // --- Setup: Initialize the Appwrite SDK ---
  const client = new sdk.Client()
    .setEndpoint(req.variables.APPWRITE_FUNCTION_ENDPOINT)
    .setProject(req.variables.APPWRITE_FUNCTION_PROJECT_ID)
    .setKey(req.variables.APPWRITE_API_KEY);

  const databases = new sdk.Databases(client);
  const users = new sdk.Users(client);

  try {
    // --- Step 1: Get the data sent from the Flutter app ---
    const { recipientId, amount, currency } = JSON.parse(req.payload);
    const senderId = req.variables.APPWRITE_FUNCTION_USER_ID; // Appwrite provides the sender's ID automatically

    if (senderId === recipientId) {
      throw new Error("You cannot send money to yourself.");
    }

    // --- Step 2: Get both the sender's and recipient's documents ---
    const [senderDoc, recipientDoc] = await Promise.all([
      databases.getDocument('afripay_db', 'users', senderId),
      databases.getDocument('afripay_db', 'users', recipientId)
    ]);

    const balanceField = currency === 'GHS' ? 'balanceGHS' : 'balanceNGN';

    // --- Step 3: Check if the sender has enough money ---
    if (senderDoc[balanceField] < amount) {
      throw new Error("Insufficient funds.");
    }

    // --- Step 4: Perform the transaction securely ---
    const newSenderBalance = senderDoc[balanceField] - amount;
    const newRecipientBalance = recipientDoc[balanceField] + amount;

    // Update both documents
    await Promise.all([
      databases.updateDocument('afripay_db', 'users', senderId, { [balanceField]: newSenderBalance }),
      databases.updateDocument('afripay_db', 'users', recipientId, { [balanceField]: newRecipientBalance })
    ]);

    // --- Step 5: Create transaction records for both users ---
    const now = new Date().toISOString();
    await Promise.all([
      // Debit record for the sender
      databases.createDocument('afripay_db', 'transactions', sdk.ID.unique(), {
        description: `Transfer to ${recipientDoc.name}`,
        amount: amount,
        type: 'debit',
        status: 'Completed',
        userId: senderId,
      }, [sdk.Permission.read(sdk.Role.user(senderId))]),
      
      // Credit record for the recipient
      databases.createDocument('afripay_db', 'transactions', sdk.ID.unique(), {
        description: `Transfer from ${senderDoc.name}`,
        amount: amount,
        type: 'credit',
        status: 'Completed',
        userId: recipientId,
      }, [sdk.Permission.read(sdk.Role.user(recipientId))])
    ]);

    console.log(`Successfully transferred ${amount} from ${senderId} to ${recipientId}`);
    res.json({ success: true, message: 'Transfer successful!' });

  } catch (error) {
    console.error("Transfer failed:", error);
    res.json({ success: false, message: error.message }, 400);
  }
};
