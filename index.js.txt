const sdk = require('node-appwrite');

// This function runs automatically AFTER a user is created.
module.exports = async (req, res) => {
  // Setup the Appwrite SDK
  const client = new sdk.Client()
    .setEndpoint(req.variables.APPWRITE_FUNCTION_ENDPOINT)
    .setProject(req.variables.APPWRITE_FUNCTION_PROJECT_ID)
    .setKey(req.variables.APPWRITE_API_KEY);

  const databases = new sdk.Databases(client);

  try {
    // Appwrite automatically provides the new user's data in the payload
    const user = JSON.parse(req.payload);

    // Create the document in the 'users' collection
    await databases.createDocument(
      '686ac6ae001f516e943e',       // Your Database ID
      '686acc5e00101633025d',       // Your 'users' Collection ID
      user.$id,                      // Use the new user's ID
      {
        'name': user.name,
        // We can't know the country here, so we default it.
        // The user can be asked to confirm their country inside the app later.
        'country': 'ghana', 
        'balanceGHS': 0.0,
        'balanceNGN': 0.0,
      },
      [
        sdk.Permission.read(sdk.Role.user(user.$id)),
        sdk.Permission.update(sdk.Role.user(user.$id)),
      ]
    );
    
    console.log(`Successfully created database document for ${user.name}`);
    res.json({ success: true });

  } catch (error) {
    console.error("Failed to create user document:", error);
    res.json({ success: false, message: error.message }, 400);
  }
};
